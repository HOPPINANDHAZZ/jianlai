    基于anime.js

    Anime.js是一个轻量的JavaScript 动画库，拥有简单而强大的API。 可对 CSS 属性、 SVG、 DOM 和JavaScript 对象进行动画。可以访问中文官网 内容导航：
    1. 可动画的目标属性
    2. 动画的目标对象
    3. 动画基础参数
    4. 方向和循环
    5. 动画赋值方式
    6. 动画关键帧
    7. 交错动画
    8. 时间轴
    9. SVG
    10. 时间曲线

    可动画的目标属性

    可使用任意CSS选择器，dom节点，js对象作为动画目标，但CSS选择器不能使用伪造元素
    CSS选择器
    复制代码 
    var animation=anime({
  targets: '.css-selector-demo1',
  translateX: 250
});
    DOM元素 /元素序列
    复制代码 
    var elements = document.querySelectorAll('.dom-node-demo2');
var animation=anime({
  targets: elements,
  translateX: 250
});
    JAVASCRIPT对象
    复制代码 
    var logEl = document.querySelector('.battery-log');
var battery = {
  charged: '0%',
  cycles: 120
}
anime({
  targets: battery,
  charged: '100%',
  cycles: 130,
  round: 1,
  easing: 'linear',
  update: function() {
    logEl.innerHTML = JSON.stringify(battery);
  }
});
    {"charged":"0%","cycles":0}

    动画的目标对象

    大多数CSS属性都会导致布局更改或重新绘制，并会导致动画不稳定。优先的是opacity和CSS transforms。
    CSS 属性
    复制代码 
    anime({
  targets: '.css-prop-demo1',
  translateX: '250px',
  backgroundColor: '#FF8F42',
  borderRadius: ['0%', '50%'],
  easing: 'easeInOutQuad'
});
    CSS3 TRANSFORMS 属性
    复制代码 
    anime({
  targets: '.css-transforms-demo1',
  translateX: 250,
  scale: 2,
  rotate: '1turn'
});
    SVG 属性
    复制代码 
    anime({
  targets: ['.svg-attributes-demo polygon', 'feTurbulence', 'feDisplacementMap'],
  points: '64 128 8.574 96 8.574 32 64 0 119.426 32 119.426 96',
  baseFrequency: 0,
  scale: 1,
  loop: true,
  direction: 'alternate',
  easing: 'easeInOutExpo'
});

    动画基础参数

    参数有持续时间、延迟、末端延迟、时间曲线函数、数字格式等，时间都是以毫秒为单位
    持续时间参数：duration
    复制代码 
    anime({
  targets: '.duration-demo',
  translateX: 250,
  duration: 3000
});
    延迟参数：delay
    复制代码 
    anime({
  targets: '.delay-demo',
  translateX: 250,
  delay: 1000
});
    末端延迟参数：endDelay
    复制代码 
    anime({
  targets: '.end-delay-demo',
  translateX: 250,
  endDelay: 500,
  direction: 'alternate'
});
    时间曲线：easing
    复制代码 
    anime({
  targets: '.easing-demo',
  translateX: 250,
  easing: 'easeInOutExpo'
});
    数字格式：round
    复制代码 
    var roundLogEl = document.querySelector('.round-log');
anime({
  targets: roundLogEl,
  innerHTML: [0, 10000],
  easing: 'linear',
  round: 10
});
    0
    特殊属性：参照文档
    复制代码 
    anime({
  targets: '.specific-prop-params-demo',
  translateX: {
    value: 250,
    duration: 800
  },
  rotate: {
    value: 360,
    duration: 1800,
    easing: 'easeInOutSine'
  },
  scale: {
    value: 2,
    duration: 1600,
    delay: 800,
    easing: 'easeInOutQuart'
  },
  delay: 300
});
    FUNCTION 参数：参照 文档
    复制代码 
    anime({
  targets: '.function-based-params-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  delay: function(el, i, l) {
    return i * 100;
  },
  endDelay: function(el, i, l) {
    return (l - i) * 100;
  }
});

    方向和循环

    可以定义动画的方向和往返次数
    方向：direction
    复制代码 
    anime({
  targets: '.dir-normal',
  translateX: 250,
  easing: 'easeInOutSine'
});

anime({
  targets: '.dir-reverse',
  translateX: 250,
  direction: 'reverse',
  easing: 'easeInOutSine'
});

anime({
  targets: '.dir-alternate',
  translateX: 250,
  direction: 'alternate',
  easing: 'easeInOutSine'
});
    循环：loop
    复制代码 
    anime({
  targets: '.loop',
  translateX: 270,
  loop: 3,
  easing: 'easeInOutSine'
});

anime({
  targets: '.loop-infinity',
  translateX: 270,
  loop: true,
  easing: 'easeInOutSine'
});

anime({
  targets: '.loop-reverse',
  translateX: 270,
  loop: 3,
  direction: 'reverse',
  easing: 'easeInOutSine'
});

anime({
  targets: '.loop-reverse-infinity',
  translateX: 270,
  direction: 'reverse',
  loop: true,
  easing: 'easeInOutSine'
});

anime({
  targets: '.loop-alternate',
  translateX: 270,
  loop: 3,
  direction: 'alternate',
  easing: 'easeInOutSine'
});

anime({
  targets: '.loop-alternate-infinity',
  translateX: 270,
  direction: 'alternate',

    动画赋值方式

    请参照 文档
    无单位数值
    复制代码 
    anime({
  targets: '.unitless-values-demo',
  translateX: 250,
  rotate: 540
});
    有单位数值
    复制代码 
    anime({
  targets: '.specific-unit-values-demo',
  width: '100%',
  easing: 'easeInOutQuad',
  direction: 'alternate',
  loop: true
});
    相对数值
    复制代码 
    var relativeEl = document.querySelector('.relative-values');
relativeEl.style.transform = 'translateX(100px)';
anime({
  targets: '.relative-values',
  translateX: {
    value: '*=2.5',
    duration: 1000
  },
  width: {
    value: '-=20px',
    duration: 1800,
    easing: 'easeInOutSine'
  },
  rotate: {
    value: '+=2turn',
    duration: 1800,
    easing: 'easeInOutSine'
  },
  direction: 'alternate'
});
    颜色
    复制代码 
    var colorsExamples = anime.timeline({
  endDelay: 1000,
  easing: 'easeInOutQuad',
  direction: 'alternate',
  loop: true
})
.add({ targets: '.color-hex',  background: '#4fab45' }, 0)
.add({ targets: '.color-rgb',  background: 'rgb(204,60,60)' }, 0)
.add({ targets: '.color-hsl',  background: 'hsl(0, 25%, 60%)' }, 0)
.add({ targets: '.color-rgba', background: 'rgba(161,119,164, .6)' }, 0)
.add({ targets: '.color-hsla', background: 'hsla(0, 30%, 10%, .2)' }, 0)
.add({ targets: '.colors-demo', translateX: 250 }, 0);
    设定动画初始值
    复制代码 
    anime({
  targets: '.from-to-values',
  translateX: [100, 250],
  delay: 500,
  direction: 'alternate',
  loop: true
});
    函数返回数值
    复制代码 
    anime({
  targets: '.function-based-values-demo',
  translateX: function(el) {
    return el.getAttribute('data-x');
  },
  translateY: function(el, i) {
    return 50 + (-50 * i);
  },
  scale: function(el, i, l) {
    return (l - i) + .25;
  },
  rotate: function() { return anime.random(-360, 360); },
  borderRadius: function() { return ['50%', anime.random(10, 35) + '%']; },
  duration: function() { return anime.random(1200, 1800); },
  delay: function() { return anime.random(0, 400); },
  direction: 'alternate',
  loop: true
});

    动画关键帧

    使用动画帧控制动画
    动画关键帧：keyframes
    复制代码 
    anime({
  targets: '.animation-keyframes-demo',
  keyframes: [
    {translateY: -40},
    {translateX: 280},
    {translateY: 40},
    {translateX: 0},
    {translateY: 0}
  ],
  duration: 4000,
  easing: 'easeOutElastic(1, .8)',
  loop: true
});
    属性关键帧
    复制代码 
    anime({
  targets: '.property-keyframes-demo',
  translateX: [
    { value: 250, duration: 1000, delay: 500 },
    { value: 0, duration: 1000, delay: 500 }
  ],
  translateY: [
    { value: -40, duration: 500 },
    { value: 40, duration: 500, delay: 1000 },
    { value: 0, duration: 500, delay: 1000 }
  ],
  scaleX: [
    { value: 4, duration: 100, delay: 500, easing: 'easeOutExpo' },
    { value: 1, duration: 900 },
    { value: 4, duration: 100, delay: 500, easing: 'easeOutExpo' },
    { value: 1, duration: 900 }
  ],
  scaleY: [
    { value: [1.75, 1], duration: 500 },
    { value: 2, duration: 50, delay: 1000, easing: 'easeOutExpo' },
    { value: 1, duration: 450 },
    { value: 1.75, duration: 50, delay: 1000, easing: 'easeOutExpo' },
    { value: 1, duration: 450 }
  ],
  easing: 'easeOutElastic(1, .8)',
  loop: true
});

    交错动画

    允许你为多个元素设置动画
    基础交错动画
    复制代码 
    anime({
  targets: '.basic-staggering-demo',
  translateX: 270,
  delay: anime.stagger(100)
});
    设定交错开始值
    复制代码 
    anime({
  targets: '.staggering-start-value-demo',
  translateX: 270,
  delay: anime.stagger(100, {start: 500})
});
    设定交错范围值
    复制代码 
    anime({
  targets: '.range-value-staggering-demo',
  translateX: 270,
  rotate: anime.stagger([-360, 360]),
  easing: 'easeInOutQuad'
});
    交错动画开始位置
    复制代码 
    anime({
  targets: '.staggering-from-demo',
  translateX: 270,
  delay: anime.stagger(100, {from: 'center'})
});
    交错动画方向
    复制代码 
    anime({
  targets: '.staggering-direction-demo',
  translateX: 270,
  delay: anime.stagger(100, {direction: 'reverse'})
});
    交错的时间曲线
    复制代码 
    anime({
  targets: '.staggering-easing-demo',
  translateX: 250,
  delay: anime.stagger(300, {easing: 'easeOutQuad'})
});
    网格交错
    复制代码 
    anime({
  targets: '.an-g1 .staggering-grid-demo',
  scale: [
    {value: .1, easing: 'easeOutSine', duration: 500},
    {value: 1, easing: 'easeInOutQuad', duration: 1200}
  ],
  delay: anime.stagger(200, {grid: [14, 5], from: 'center'})
});
    网格交错方向
    复制代码 
    anime({
  targets: '.an-g2 .staggering-grid-demo',
  translateX: anime.stagger(10, {grid: [14, 5], from: 'center', axis: 'x'}),
  translateY: anime.stagger(10, {grid: [14, 5], from: 'center', axis: 'y'}),
  rotateZ: anime.stagger([0, 90], {grid: [14, 5], from: 'center', axis: 'x'}),
  delay: anime.stagger(200, {grid: [14, 5], from: 'center'}),
  easing: 'easeInOutQuad'
});

    时间轴

    可让你将多个动画同步在一起
    基础时间轴
    复制代码 
    var tl = anime.timeline({
  easing: 'easeOutExpo',
  duration: 750
});

tl
.add({
  targets: '.basic-timeline-demo1',
  translateX: 250,
})
.add({
  targets: '.basic-timeline-demo2',
  translateX: 250,
})
.add({
  targets: '.basic-timeline-demo3',
  translateX: 250,
});
    时间轴偏移量
    复制代码 
    var tl = anime.timeline({
  easing: 'easeOutExpo',
  duration: 750
});
tl
.add({
  targets: '.offsets-demo1',
  translateX: 250,
})
.add({
  targets: '.offsets-demo2',
  translateX: 250,
}, '-=600')
.add({
  targets: '.offsets-demo3',
  translateX: 250,
}, 0);
    参数继承
    复制代码 
    var tl = anime.timeline({
  targets: '.params-inheritance-demo',
  delay: function(el, i) { return i * 200 },
  duration: 500,
  easing: 'easeOutExpo',
  direction: 'alternate',
  loop: true
});
tl
.add({
  translateX: 250,
  easing: 'spring',
})
.add({
  opacity: .5,
  scale: 2
})
.add({
  targets: '.params-inheritance-demo .params-inheritance-demo2',
  rotate: 180
})
.add({
  translateX: 0,
  scale: 1
});

    SVG

    动画一个元素，使之相对于SVG路径的x，y和角度值运动。
    运动路径动画
    复制代码 
    var path = anime.path('.motion-path-demo path');
anime({
  targets: '.motion-path-demo .el',
  translateX: path('x'),
  translateY: path('y'),
  rotate: path('angle'),
  easing: 'linear',
  duration: 2000,
  loop: true
});
    变形动画
    复制代码 
    anime({
  targets: '.morphing-demo .polymorph',
  points: [
    { value: [
      '70 24 119.574 60.369 100.145 117.631 50.855 101.631 3.426 54.369',
      '70 41 118.574 59.369 111.145 132.631 60.855 84.631 20.426 60.369']
    },
    { value: '70 6 119.574 60.369 100.145 117.631 39.855 117.631 55.426 68.369' },
    { value: '70 57 136.574 54.369 89.145 100.631 28.855 132.631 38.426 64.369' },
    { value: '70 24 119.574 60.369 100.145 117.631 50.855 101.631 3.426 54.369' }
  ],
  easing: 'easeOutQuad',
  duration: 2000,
  loop: true
});
    画线动画
    复制代码 
    anime({
  targets: '.line-drawing-demo .lines path',
  strokeDashoffset: [anime.setDashoffset, 0],
  easing: 'easeInOutSine',
  duration: 1500,
  delay: function(el, i) { return i * 250 },
  direction: 'alternate',
  loop: true
});

    时间曲线

    函数请访问官方文档
    匀速
    复制代码 
    anime({
  targets: '.linear-easing-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  easing: 'linear'
});
    三次贝塞尔
    复制代码 
    anime({
  targets: '.cubic-bezier-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  easing: 'cubicBezier(.5, .05, .1, .3)'
})
    弹簧
    复制代码 
    anime({
  targets: '.spring-physics-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  easing: 'spring(1, 80, 10, 0)'
})
    弹跳
    复制代码 
    anime({
  targets: '.elastic-easing-demo',
  translateX: 250,
  easing: function(el) {
    return el.getAttribute('data-ease');
  },
  duration: 1000
})
    台阶式
    复制代码 
    anime({
  targets: '.step-easing-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  easing: 'steps(5)'
})
    自定义
    复制代码 
    anime({
  targets: '.custom-easing-demo',
  translateX: 250,
  direction: 'alternate',
  loop: true,
  duration: 2000,
  easing: function(el, i, total) {
    return function(t) {
      return Math.pow(Math.sin(t * (i + 1)), total);
    }
  }
});
